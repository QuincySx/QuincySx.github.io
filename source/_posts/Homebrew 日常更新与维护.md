```
brew update //更新 Homebrew

brew install #name //安装软件

brew outdated //查看那些软件可更新

brew upgrade //更新所有软件

brew upgrade #name //更新指定软件

brew cleanup //清理旧版本软件

brew cleanup #name //清理指定软件的旧版本

brew cleanup -n //查看可清理的软件
```
