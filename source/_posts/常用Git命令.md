```
  //添加远程仓库
  git remote add origin git@github.com:xx.git
  
  //删除远端分支
  git push origin --delete <branchName>
  
  //添加 Tag
  git tag v1.0
  
  //删除远端 Tag
  git push origin --delete tag <tagname>
  
  //暂存文件
  git stash
  
  //拿出暂存文件
  git stash pop
```
